CREATE TABLE Employee (
	employeeID		INT(4) NOT NULL PRIMARY KEY AUTO_INCREMENT,
	firstName		Varchar(50) NOT NULL,
	lastName		Varchar(50) NOT NULL,
	department		Varchar(50),
	phone			Char(10),
	email  			Varchar(100),
	gender			char(1),
	manager			char(1)
	);
